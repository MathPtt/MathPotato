1use uuid::Uuid;
2
3use crate::parser::parser_error::ParseError;
4
5use super::ast_node_types_enum::AstNodeType;

7use super::infix_operation_enum::InfixOperationType;

9#[derive(Debug, Clone)]
10pub struct InfixOperationAstNode {
11    operation_type: InfixOperationType,
12    left_id: Uuid,
13    left_type: AstNodeType,
14    right_id: Uuid,
15    right_type: AstNodeType,
16}
17
18impl InfixOperationAstNode {
19    pub fn new_with_type_and_left_child_node(
20        operation_type: InfixOperationType,
21        left_type: AstNodeType,
22        left: Uuid,
23    ) -> InfixOperationAstNode {
24        InfixOperationAstNode {
25            operation_type,
26            left_type,
27            left_id: left,
28            right_id: Uuid::nil(),
29            right_type: AstNodeType::None,
30        }
31    }
32    pub fn get_operation_type(&self) -> InfixOperationType {
33        self.operation_type.clone()
34    }
35
36    pub fn get_left_node_id_and_type(&self) -> Option<(Uuid, AstNodeType)> {
37        if self.left_id == Uuid::nil() {
38            None
39        } else {
40            Some((self.left_id, self.left_type.clone()))
41        }
42    }
43
44    pub fn get_right_node_id_and_type(&self) -> Option<(Uuid, AstNodeType)> {
45        if self.right_id == Uuid::nil() {
46            None
47        } else {
48            Some((self.right_id, self.right_type.clone()))
49        }
50    }
51
52    pub fn get_right_node(&self) -> Option<(Uuid, AstNodeType)> {
53        if self.right_id == Uuid::nil() {
54            None
55        } else {
56            Some((self.right_id, self.right_type.clone()))
57        }
58    }
59
60    /// Checks if the InfixOperationAstNode node is in the right state.
61    ///
62    /// # Concept
63    ///
64    /// An InfixOperationAstNode left and right side is occupied in the following order:
65    /// - Left side (since we read from the left to right)
66    /// - Right side
67    pub(crate) fn check_if_left_empty_right_occupied(&self) -> Result<(), ParseError> {
68        if self.left_id != Uuid::nil()
69            && self.left_type != AstNodeType::None
70            && self.right_id != Uuid::nil()
71            && self.right_type != AstNodeType::None
72        {
73            Err(ParseError::new(String::from("InfixOperationAstNode is in an impossible state. It has its left side empty, but right side occupied.")))
74        } else {
75            Ok(())
76        }
77    }
78
79    pub(crate) fn is_left_occupied(&self) -> bool {
80        self.left_type != AstNodeType::None && self.left_id != Uuid::nil()
81    }
82
83    pub(crate) fn is_right_occupied(&self) -> bool {
84        self.right_type != AstNodeType::None && self.right_id != Uuid::nil()
85    }
86
87    pub(crate) fn add_i32node_to_the_right(
88        &mut self,
89        id: Uuid,
90    ) -> Result<(Uuid, InfixOperationAstNode), ParseError> {
91        if self.right_id != Uuid::nil() && self.right_type != AstNodeType::None {
92            Err(ParseError::new(format!(
93                "The right side is not empty. It is occupied by id: {:#?} and type: {:#?}",
94                self.right_id, self.right_type
95            )))
96        } else {
97            self.right_id = id;
98            self.right_type = AstNodeType::I32AstNode;
99            Ok((id, self.clone()))
100        }
101    }
102}
