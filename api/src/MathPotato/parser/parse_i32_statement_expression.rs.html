19/// Parses integer, i32, statement expression based on the tokens provided by the lexer.
20///
21/// # Parameters
22///
23/// * `i` - start character of processing
24/// * `tokens` - PotatoTokens
25/// * `nodes` - Abstract Syntax Tree
26///
27/// # Returns
28///
29/// Abstract Syntax Tree, `VariableValueExpressionNode` representing the processed tokens.
30///
31/// # Examples
32///
33/// The method processes whatever is between `=` and `;`;
34/// ```
35/// i32 asd = 5;
36/// i32 foo = 4 + 4;
37/// ```
38pub fn parse_i32_statement_expression(
39    i: usize,
40    tokens: Vec<PotatoToken>,
41    mut ast: MathPotatoAstTree,
42) -> Result<MathPotatoAstTree, ParseError> {
43    match tokens.get(i).cloned().ok_or_else(|| error_message(i)) {
44        Err(e) => panic!("{}", e),
45
46        Ok(token) => {
47            match ast.get_continuation_node_id_and_type() {
48                // this means that we are processing the first character!
49                None => {
50                    match token.token_type {
51                        PotatoTokenTypes::LiteralIntegerValue => {
52                            // this is the case when we right after the `=` sign and the expression
53                            // tree is empty
54                            let recorded_node = ast
55                                .put_i32_ast_node(I32AstNode::new_with_value(parse_literal_to_i32(
56                                    &token,
57                                )))
58                                .unwrap_or_else(|e| {
59                                    panic!("Adding I32AstNode has failed. Details: {:#?}", e)
60                                });
61
62                            let _ = ast
63                                .add_root_node_id_and_type(
64                                    recorded_node.id,
65                                    AstNodeType::I32AstNode,
66                                )
67                                .unwrap_or_else(|r| {
68                                    panic!(
69                                        "Adding root node details to AST failed. Details: {:#?}",
70                                        r
71                                    )
72                                });
73                            parse_i32_statement_expression(i + 1, tokens, ast)
74                        }
75                        PotatoTokenTypes::SignAssignment => {
76                            parse_i32_statement_expression(i + 1, tokens, ast)
77                        }
78                        PotatoTokenTypes::OperationDivision => {
79                            parse_i32_statement_expression(i + 1, tokens, ast)
80                        }
81                        PotatoTokenTypes::OperationAddition => {
82                            panic!(
83                                "Syntax error! The first character of an expression cannot be +!"
84                            )
85                        }
86                        PotatoTokenTypes::KeywordI32 => {
87                            parse_i32_statement_expression(i + 1, tokens, ast)
88                        }
89                        PotatoTokenTypes::SignCloseParentheses => {
90                            parse_i32_statement_expression(i + 1, tokens, ast)
91                        }
92                        PotatoTokenTypes::LiteralValueVariableIdentifier => {
93                            parse_i32_statement_expression(i + 1, tokens, ast)
94                        }
95                        PotatoTokenTypes::SignOpenParentheses => {
96                            parse_i32_statement_expression(i + 1, tokens, ast)
97                        }
98                        PotatoTokenTypes::SignSemicolon => Ok(ast),
99                        PotatoTokenTypes::None => todo!(),
100                    }
101                }
102                Some(cont_node_details) => {
103                    // once we matched the actual token type
104                    // we are going to match on continuation node too.
105
106                    match token.token_type {
107                        PotatoTokenTypes::LiteralIntegerValue => {
108                            match cont_node_details.1 {
109                                AstNodeType::I32AstNode => {
110                                    // this is a syntax error, since two number type cannot follow each other
111                                    Err(ParseError::new(String::from(
112                                        "Syntax error! Two number type cannot follow each other!",
113                                    )))
114                                }
115                                AstNodeType::InfixOperationAstNode => {
116                                    // this happens when we are right after the `+` sign in the
117                                    // following example:
118                                    // `i32 variable_name = 1 + 3;`
119                                    let mut cont_node = ast
120                                        .get_infix_node_by_id(cont_node_details.0)
121                                        .unwrap_or_else(|| {
122                                            panic!(
123                                                "No InfixOperationAstNode with id {:#?}",
124                                                cont_node_details.0
125                                            )
126                                        });
127                                    cont_node
128                                        .1
129                                        .check_if_left_empty_right_occupied()
130                                        .unwrap_or_else(|e| panic!("{:#?}", e));
131                                    let i32node = I32AstNode::new_value_parent_id_and_type(
132                                        parse_literal_to_i32(&token),
133                                        AstNodeType::InfixOperationAstNode,
134                                        cont_node.0,
135                                    );
136                                    let i32node_recorded = ast
137                                        .put_i32_ast_node(i32node)
138                                        .unwrap_or_else(|e| panic!("{:#?}", e));
139                                    cont_node
140                                        .1
141                                        .add_i32node_to_the_right(i32node_recorded.id)
142                                        .unwrap_or_else(|e| panic!("{:#?}", e));
143                                    ast.update_infix_node_by_id(cont_node.0, cont_node.1).unwrap_or_else(|e|panic!("Error happened while persisting updated InfixOperationAstNode node. {:#?}", e));
144                                    parse_i32_statement_expression(i + 1, tokens, ast)
145                                }
146                                AstNodeType::None => {
147                                    panic!("we have a cont node, but the type is none")
148                                }
149                            }
150                        }
151                        PotatoTokenTypes::SignAssignment => {
152                            parse_i32_statement_expression(i + 1, tokens, ast)
153                        }
154                        PotatoTokenTypes::SignCloseParentheses => {
155                            parse_i32_statement_expression(i + 1, tokens, ast)
156                        }
157                        PotatoTokenTypes::SignOpenParentheses => {
158                            parse_i32_statement_expression(i + 1, tokens, ast)
159                        }
160                        PotatoTokenTypes::OperationDivision => {
161                            parse_i32_statement_expression(i + 1, tokens, ast)
162                        }
163                        PotatoTokenTypes::OperationAddition => {
164                            match cont_node_details.1 {
165                                AstNodeType::I32AstNode => {
166                                    // this means we are at the `+` in the following example
167                                    // `i32 variable_name = 1 + 2`
168                                    // we are going to create a infix operation node and
169                                    // the continuation node will be a child to it
170                                    // and the new node will be the parent of the continuation node
171                                    let mut cont_node = ast
172                                        .get_i32_node_by_id(cont_node_details.0)
173                                        .unwrap_or_else(|e| {
174                                            panic!(
175                                                "There is no i32 continuation node with {}. Error: {:#?}",
176                                                cont_node_details.0,
177                                                e
178                                            )
179                                        });
180
181                                    let infix_node =
182                                        InfixAstNodeInternal::new_with_type_and_left_child_node(
183                                            InfixOperationType::Addition,
184                                            cont_node_details.1,
185                                            cont_node_details.0,
186                                        );
187                                    let recorded_infix_node =
188                                        ast.put_infix_node(infix_node).unwrap_or_else(|err| {
189                                            panic!(
190                                                "Error while adding InfixNode to AST. Error: {:#?}",
191                                                err
192                                            )
193                                        });
194                                    cont_node.parent_type = AstNodeType::InfixOperationAstNode;
195                                    cont_node.parent_id = recorded_infix_node.0;
196                                    let _ = ast.update_i32_node(
197                                                        cont_node_details.0,
198                                                        cont_node,
199                                                    ).unwrap_or_else(|r|
200                                            panic!("Updating the I32AstNode with the added InfixNode details failed. Details: {:#?}", r));
201                                    let _ = ast
202                                    .update_root_node_id_and_type(recorded_infix_node.0, AstNodeType::InfixOperationAstNode)
203                                        .unwrap_or_else(|e|panic!("Error happened while updated root node id and type. Error: {:#?}", e));
204
205                                    parse_i32_statement_expression(i + 1, tokens, ast)
206                                }
207                                AstNodeType::InfixOperationAstNode => {
208                                    panic!("Syntax error! Infix operation node cannot follow another one.")
209                                }
210                                AstNodeType::None => {
211                                    panic!("The continuation node type is None.")
212                                }
213                            }
214                        }
215                        PotatoTokenTypes::KeywordI32 => {
216                            parse_i32_statement_expression(i + 1, tokens, ast)
217                        }
218                        PotatoTokenTypes::LiteralValueVariableIdentifier => {
219                            parse_i32_statement_expression(i + 1, tokens, ast)
220                        }
221                        PotatoTokenTypes::SignSemicolon => Ok(ast),
222                        PotatoTokenTypes::None => todo!(),
223                    }
224                }
225            }
226        }
227    }
228}

230fn error_message(i: usize) -> String {
231    format!("There is no character at {}", i)
232}

234fn parse_literal_to_i32(t: &PotatoToken) -> i32 {
235    let result: Result<i32, _> = t.literal_value.parse();
236    match result {
237        Ok(v) => v,
238        Err(e) => {
239            panic!(
240                "Error while parsing literal value to i32. Error message is {}",
241                e
242            );
243        }
244    }

262    fn value_infix_and_value() {
263        // case: `3 + 3;`
264        // arrange
265        let input = String::from("1 + 2;");
266        let lexed_input = lexing(&input);
267        let input_ast = MathPotatoAstTree::new();
268
269        // action
270        let result = parse_i32_statement_expression(0, lexed_input, input_ast)
271            .unwrap_or_else(|e| panic!("There is no result: {:#?}", e));
272
273        // assert
274        assert_eq!(
275            result.i32_node_count(),
276            2,
277            "sout: {}, expected: {}",
278            result.i32_node_count(),
279            2
280        );
281        assert_eq!(
282            result.infix_tree_len(),
283            1,
284            "sout: {}, expected: {}",
285            result.infix_tree_len(),
286            1
287        );
288
289        // continuation node checks
290        let continuation_node_id_and_type = result
291            .get_continuation_node_id_and_type()
292            .unwrap_or_else(|| panic!("There is no continuation node!"));
293        assert_eq!(
294            continuation_node_id_and_type.1,
295            AstNodeType::I32AstNode,
296            "The continuation node has to be an I32AstNode"
297        );
298        let _ = result
299            .get_i32_node_by_id(continuation_node_id_and_type.0)
300            .unwrap_or_else(|e| panic!("There is no i32 continuation node by id. Error: {:#?}", e));
301
302        // root node checks
303        let root_node_id = result
304            .get_root_node_id()
305            .unwrap_or_else(|| panic!("There is no root node id!"));
306        assert_eq!(
307            result.get_root_node_type(),
308            AstNodeType::InfixOperationAstNode,
309            "Root node must be: {:#?}, but it was {:#?}",
310            AstNodeType::InfixOperationAstNode,
311            result.get_root_node_type()
312        );
313        let root_node = result
314            .get_root_node_infix(root_node_id)
315            .unwrap_or_else(|e| panic!("{:#?}", e));
316        assert!(
317            root_node.1.is_left_occupied(),
318            "The left node of the root InfixOperationAstNode must be occupied at this point.",
319        );
320        assert!(
321            root_node.1.is_right_occupied(),
322            "The left node of the root InfixOperationAstNode must be occupied at this point.",
323        );
324        let left_children_id_and_type = root_node
325            .1
326            .get_left_node_id_and_type()
327            .unwrap_or_else(|| panic!("There is no left children."));
328        assert_eq!(
329            left_children_id_and_type.1,
330            AstNodeType::I32AstNode,
331            "Children node has to be {:#?}",
332            AstNodeType::I32AstNode
333        );
334        let left_children = result
335            .get_i32_node_by_id(left_children_id_and_type.0)
336            .unwrap_or_else(|e| panic!("No left children. Error: {:#?}", e));
337        assert_eq!(left_children.value, 1, "The left children value must be 1");
338        assert_eq!(
339            left_children.parent_id, root_node.0,
340            "The parent id must be equal to root node id."
341        );
342        assert_eq!(
343            left_children.parent_type,
344            result.get_root_node_type(),
345            "The parent type must be equal to root node type."
346        );
347
348        let right_children_id_and_type = root_node
349            .1
350            .get_right_node_id_and_type()
351            .unwrap_or_else(|| panic!("There is no left children."));
352        assert_eq!(
353            right_children_id_and_type.1,
354            AstNodeType::I32AstNode,
355            "Children node has to be {:#?}",
356            AstNodeType::I32AstNode
357        );
358        let right_children = result
359            .get_i32_node_by_id(right_children_id_and_type.0)
360            .unwrap_or_else(|e| panic!("No right children. Error: {:#?}", e));
361        assert_eq!(
362            right_children.value, 2,
363            "The right children value must be 2"
364        );
365        assert_eq!(
366            right_children.parent_id, root_node.0,
367            "The parent id must be equal to root node id."
368        );
369        assert_eq!(
370            right_children.parent_type,
371            result.get_root_node_type(),
372            "The parent type must be equal to root node type."
373        );
374    }

377    fn value_and_infixoperation() {
378        // arrange
379        let input = String::from("3 +;");
380        let lexed_input = lexing(&input);
381        let input_ast = MathPotatoAstTree::new();
382
383        // action
384        let result = parse_i32_statement_expression(0, lexed_input, input_ast)
385            .unwrap_or_else(|e| panic!("There is no result: {:#?}", e));
386
387        // assert
388        assert_eq!(result.i32_node_count(), 1);
389        assert_eq!(result.infix_tree_len(), 1);
390
391        // continuation node checks
392        let continuation_node_id_and_type = result
393            .get_continuation_node_id_and_type()
394            .unwrap_or_else(|| panic!("There is no continuation node!"));
395        assert_eq!(
396            continuation_node_id_and_type.1,
397            AstNodeType::InfixOperationAstNode
398        );
399        let cont_node = result
400            .get_infix_node_by_id(continuation_node_id_and_type.0)
401            .unwrap_or_else(|| panic!("There is no continuation node by id"));
402        assert_eq!(cont_node.0, continuation_node_id_and_type.0);
403        assert_eq!(
404            cont_node.1.get_operation_type(),
405            InfixOperationType::Addition
406        );
407
408        // root node checks
409        let root_node_id = result
410            .get_root_node_id()
411            .unwrap_or_else(|| panic!("There is no root node id!"));
412        assert_eq!(
413            result.get_root_node_type(),
414            AstNodeType::InfixOperationAstNode
415        );
416        let root_node = result
417            .get_infix_node_by_id(root_node_id)
418            .unwrap_or_else(|| panic!("There is no root node!"));
419        assert_eq!(
420            root_node.1.get_operation_type(),
421            InfixOperationType::Addition
422        );
423        let root_node_right_child_id_and_type = root_node
424            .1
425            .get_left_node_id_and_type()
426            .unwrap_or_else(|| panic!("There is no children for root node"));
427        assert_eq!(root_node_right_child_id_and_type.1, AstNodeType::I32AstNode);
428
429        // child node
430        let child_node = result
431            .get_i32_node_by_id(root_node_right_child_id_and_type.0)
432            .unwrap_or_else(|e| {
433                panic!(
434                    "There is no node under the provided root node's child item node. Error: {:#?}",
435                    e
436                )
437            });
438        assert_eq!(child_node.value, 3);
439        assert_eq!(child_node.parent_type, AstNodeType::InfixOperationAstNode);
440        assert_eq!(child_node.parent_id, root_node_id);
441    }

444    fn value_only() {
445        // arrange
446        let input = String::from("3;");
447        let lexed_input = lexing(&input);
448        let input_ast = MathPotatoAstTree::new();
449
450        // action
